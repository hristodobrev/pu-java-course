CREATE TABLE PEOPLE(
    ID IDENTITY PRIMARY KEY,
    FIRST_NAME NVARCHAR(255) NOT NULL,
    LAST_NAME NVARCHAR(255) NOT NULL,
    GENDER NVARCHAR(10) NOT NULL,
    AGE INT NOT NULL CHECK (AGE>0),
    SALARY DECIMAL(8,2) NOT NULL
);
